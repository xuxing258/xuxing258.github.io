http模块

Node.js提供了内置的http模块，用于创建HTTP服务器和处理HTTP请求和响应。你可以使用http模块来构建基于HTTP协议的服务器应用程序

基础使用

const http = require('http');

// 创建服务器对象
const server = http.createServer((req, res) => {
  // 处理请求
  res.statusCode = 200; // 设置响应状态码
  res.setHeader('Content-Type', 'text/plain'); // 设置响应头
  res.end('Hello, World!'); // 发送响应数据
});

// 监听端口
server.listen("3000", () => {
  console.log(`3000端口执行`);
});
  • server.listen([prot[, callback]])
  • prot : 端口号 [0,65536]
  • 当浏览器访问 localhost:端口号 时执行 http.createServer的回调函数
//当浏览器请求 localhost:端口号  触发回调函数
let server = http.createServer(function(request,response){
  //设置响应头基础信息  服务器返回的数据不是乱码  
  response.setHeader('Content-Type', 'text/html;charset=utf-8')
  response.end(333)
})
  • request 请求体
  • response 响应体

response方法

  • end()

    • 参数为: buffer或string

  • setHeader(name,value)

    • 响应头设置单个响应头的值

request 属性

  • url 请求的 URL 字符串信息

配置多个请求

前端的post数据存在请求体中的, 并且是分批次数传输,最有超大数才会显示,

接收post携带数据会触发req的data事件 , 接收数据完毕会触req的end事件

res.setHeader() 的响应的格式 会最终合并到 res.writeHead()里面

// http模块
const http = require("http");
// 文件系统模块
const fs = require("fs");
// url 地址栏模块
const url = require("url")
//querystring 模块提供用于解析和格式化 URL 查询字符串的实用工具
const qs = require("querystring");
// 开启服务器环境
let server = http.createServer(function (req,res) {
  	// url模块将地址序列对象话
    const  {pathname} = url.parse(req.url)
		// 判断路径
    if (pathname == "/"){
        fs.readFile("./123.html",function(err,data){
            if(err)return;
            res.end(data)
        })
      // 请求地址为/xuxin是触发 post请求
    }else if (pathname == "/xuXin"){
        let result = ""
        // 接收数据触发
        req.on("data",function (chunk) {
            result +=  chunk
        });
        // 数据接收完毕 触发函数
        req.on("end",function (chunk) {
            // querystring模块 解决url数据
            console.log(qs.parse(result))
            // 使用res.setHeader() 的响应的格式 会最终合并到 res.writeHead()里面
            res.writeHead(200,{
                'Content-Type': 'text/html;charset=utf-8'
            })
            res.end("数据响应完毕")
        })

    }else{
        res.end("404")
    }
});

server.listen(3000,function(){
    console.log("3000端口开启")
})
