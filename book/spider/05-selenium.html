selenium介绍

Selenium 是一个强大的浏览器自动化工具, 广泛用于网页测试和爬取动态加载内容, (主要是用于自动化测试)

自动化是指使用技术手段模拟人工，执行重复性任务,

安装 pip install selenium

官网: https://www.selenium.dev/zh-cn/documentation/webdriver/browsers/chrome/

浏览器的启动

webdriver.Chrome() webdriver.Edge() webdriver.Firefox() webdriver.Safari()

# 引入webdriver驱动,启动浏览器
from selenium import webdriver 
# 启动浏览器 需要打开对应窗口
driver = webdriver.Chrome()     	# 可以设置其他浏览器
# 打开网页  打开后浏览器会自动关闭
driver.get("https://www.baidu.com")
# 可以设置input()输入内容 停止浏览器关闭
input("请输入内容") # 输入内容之后,会自动关闭
  • driver.page_source 获取网页源代码
  • driver.close() 关闭页面
  • driver.quit() 退出浏览器
  • driver.title 获取网页的title标签信息
  • driver.current_url 获取当前网页的url

浏览器配置信息

options = webdriver.ChromeOptions() 
# 两种使用方式功能一致
from selenium.webdriver.chrome.options import Options
options = Options()
# edge浏览器: webdriver.EdgeOptions()
  • ("detach", True) 保持浏览器的打开状态
  • ('excludeSwitches', ['enable-automation'])
    • 隐藏浏览器的自动化痕迹，避免被网站检测到是自动化程序在操作
  • --start-maximized 最大化窗口
  • --no-sandbox 禁用沙盒模式, 提高浏览器权限
from selenium import webdriver 
# 设置浏览器配置信息  
options = webdriver.ChromeOptions() 
# 保持浏览器的打开状态
options.add_experimental_option("detach", True)
# 隐藏自动化提示栏，并禁用部分自动化检测机制
options.add_experimental_option('excludeSwitches', ['enable-automation'])
# 最大化窗口
options.add_argument("--start-maximized") 
# 沙盒模式
options.add_argument("--no-sandbox")     
# 传入配置信息
driver = webdriver.Chrome(options=options)
driver.get("https://www.baidu.com")
# 获取网页的源码
print(driver.page_source)
print(driver.title)

获取元素

百度首页演示: https://www.baidu.com/index.htm

  • 浏览器对象.find_element(By.type, "value")
    • type: 选择对应方式定位元素
      • 示例: By.ID 或 By.CLASS_NAME
    • value: 是type方式对应的类名或id名等等
  • 元素节点.get_attribute("outerHTML") 获取元素整体内容
  • 元素节点.text 获取元素文本内容
# 导入By 根据查找网页元素
from selenium.webdriver.common.by import By
driver = webdriver.Chrome()
driver.get('https://www.baidu.com')

# 通过ID定位
element = driver.find_element(By.ID, "id_value")
# 获取element元素完整内容
html_content = element.get_attribute("outerHTML")
  • 通过class名定位
    • driver.find_element(By.CLASS_NAME, "class_name")
  • 通过HTML标签名定位
    • driver.find_element(By.TAG_NAME, "input")
  • 通过CSS选择器定位
    • driver.find_element(By.CSS_SELECTOR, "div.login > input")
  • 通过XPath定位
    • driver.find_element(By.XPATH, "//input[@name='email']")
# 通过链接文本定位（完全匹配）
<a href="/about">About Us</a>
element = driver.find_element(By.LINK_TEXT, "About Us")
# 通过链接文本定位（部分匹配）
<a href="/about">About Us</a>
element = driver.find_element(By.PARTIAL_LINK_TEXT, "About")

# 通过name属性定位
<input name="email" type="text">
driver.find_element(By.NAME, "email")

from selenium.webdriver import ActionChains

# 产生一个动作执行器
a = driver.find_element(By.ID, "draggable")

actions.perform()  #执行

等待

WebDriverWait通常用于等待网页某个元素加载完毕后进行后续操作，避免出现异常

from selenium import webdriver
from selenium.webdriver.common.by import By 
# 等待页面加载完毕，寻找某些元素 
from selenium.webdriver.support.wait import WebDriverWait 
# 用于与 WebDriverWait 配合使用，实现智能等待页面元素的各种状态
from selenium.webdriver.support import expected_conditions as EC

browser = webdriver.Chrome()
browser.get('https://www.taobao.com/')

# 创建等待对象，设置最长等待时间为 10秒
wait = WebDriverWait(browser,  timeout=10)
# 使用wait查询 会等待元素加载成功,最长等待10秒, 超过时间未获取则报错
input = wait.until(EC.presence_of_element_located((By.ID,'J_SiteFooter')))

等待条件: 在文档的额外功能, 期望状态位置

# 创建等待对象，设置最长等待时间为 10秒
wait = WebDriverWait(driver, timeout=10)

# 等待元素在DOM中存在且可见（宽度/高度>0，非隐藏）
visible_element = wait.until(EC.visibility_of_element_located((By.XPATH, "//div")))

# 检查元素是否出现在DOM中
hidden_element = wait.until(EC.presence_of_element_located((By.CLASS_NAME, "item")))

# 等待元素可见且可点击（Enabled状态） 用于:提交按钮、超链接等
login_button = wait.until(EC.element_to_be_clickable((By.LINK_TEXT, "登录")))

# 验证文本内容
success_msg = wait.until(EC.text_to_be_present_in_element((By.ID, "status"), "成功"))

# 等待元素不可见或从DOM中移除（常用于等待加载动画消失）
wait.until(EC.invisibility_of_element_located((By.ID, "loading")))

浏览器配置

有效降低被网站检测为自动化的风险，适用于爬虫和自动化测试场景

参数作用典型应用场景
--start-maximized启动时最大化窗口避免元素因窗口太小导致不可见
--headless=new无头模式（不显示GUI界面）服务器环境或不需要可视化时节省资源
--user-data-dir=/path/to/profile加载指定用户数据目录（保存Cookies/扩展/历史记录）复用登录状态或自定义浏览器环境
--incognito隐身模式（不读取历史记录，新建临时会话）确保每次启动都是干净环境
--disable-gpu禁用GPU加速（旧版无头模式可能需要）解决无头模式下的兼容性问题
--window-size=1200,800设置初始窗口大小（宽×高）需要固定窗口尺寸时使用
--disable-extensions禁用所有扩展避免扩展干扰自动化操作
--no-sandbox禁用沙盒模式（Linux服务器可能需要）解决权限不足导致的崩溃问题
from selenium.webdriver.chrome.options import Options
options = Options()
options.add_argument("--start-maximized")       # 最大化窗口
options.add_argument("--headless=new")          # 无头模式  不显示浏览器

# 存储用户数据目录 (保存cookies、缓存等)  使其访问时, 能保持登录状态
options.add_argument("user-data-dir=D:/授课/1-15肥虫班/Day8/jd/data") # 必须为绝对路径，且目录需已存在

driver = webdriver.Chrome(options)
driver.get('https://www.baidu.com')

def spider():

    options = webdriver.ChromeOptions()
    # 配置禁止自动化检测, 隐藏浏览器自动化痕迹
    options.add_experimental_option('excludeSwitches', ['enable-automation'])
    # 禁止浏览器自动关闭
    options.add_experimental_option("detach", True)
    # 设置用户信息进行缓存
    options.add_argument("user-data-dir=C:/Users/15116/Desktop/lianxi/userdata")
    # 通过webdriver启动谷歌浏览器
    driver = webdriver.Chrome(options=options)
    # 打开淘宝网址
    driver.get("https://www.taobao.com")
    # 设置浏览器等待程序 等待十秒钟
    wait = WebDriverWait(driver, 10)
    # 设置等待元素存在并且可见
    dom = wait.until(EC.presence_of_element_located((By.ID, "q")))
    # 输入内容并且回车,进行跳转页面
    dom.send_keys("美食")
    dom.send_keys(Keys.ENTER)

    driver.switch_to.window(driver.window_handles[1])
    # 获取网页源码, 并传递到get_html中进行解析, 获取数据
    time.sleep(10)
    set_html_content(driver)
    # 设置浏览器不关闭
    # input("请输入")
    pass


# 设置搜索内容
def set_html_content(driver):
    global index
    index += 1
    print(f"第{index}页")
    # 控制页面滚动到最下方
    js = 'window.scrollTo(0,document.documentElement.scrollHeight)'
    driver.execute_script(js)
    time.sleep(5)
    code_info = etree.HTML(driver.page_source)
    # 可以通过lxml 或 soup进行解析获取数据
    dom_a_href = code_info.xpath('//div[@id="content_items_wrapper"]/div/a/@href')
    dom_title_txt = code_info.xpath('//div[@id="content_items_wrapper"]/div/a//div[@class="title--ASSt27UY "]/span//text()')
    dom_price = code_info.xpath('//div[@id="content_items_wrapper"]/div/a//div[@class="innerPriceWrapper--aAJhHXD4"]//text()')

    for dom_a, dom_title, price in zip(dom_a_href, dom_title_txt, dom_price):
        # 详情地址: 有些数据是没有添加https的
        if not dom_a.startswith('http'):
            dom_a = "https:" + dom_a

        print(dom_title, price,)
        # 获取当前浏览器页面
        cur_driver = driver.current_window_handle
        # 进行跳转
        driver.execute_script(f"window.open('{dom_a}', '_blank')")
        time.sleep(5)
        # 切换浏览器对象信息
        driver.switch_to.window(driver.window_handles[2])
        time.sleep(3)
        name_title = driver.find_element(By.XPATH, '//*[@id="left-content-area"]/div[1]/a/div/div[2]/div[1]/span')
        print(name_title.get_attribute("innerText"))
        print("===============================================")
        driver.close()
        driver.switch_to.window(cur_driver)
        time.sleep(5)

    # 需要进行等待页面的加载
    time.sleep(3)
    # 等待获取到下一页的按钮
    wait = WebDriverWait(driver, 15)
    dom_next = wait.until(EC.presence_of_element_located((By.XPATH, '//*[@id="search-content-leftWrap"]/div[2]/div[6]/div/div/button[2]')))
    # 如果存在disabled属性,就
    if not dom_next.get_attribute("disabled"):
        print("下一页")
        dom_next.click()
        # 获取当前的浏览器信息
        set_html_content(driver)
    else:
        print("程序结束")
    pass
