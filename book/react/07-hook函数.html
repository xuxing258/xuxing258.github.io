useCallback

是一个允许你在多次渲染中缓存函数的API const cachedFn = useCallback(fn, dependencies)

  1. useCallback 不会执行参数函数，而是返回函数存储版本函数
  2. 防止组件重新渲染，导致方法被重新创建 ，useCallback起到缓存作用*;*
    • 只有依懒参数变化才会返回新的函数
  3. 依懒项
  • [] 空数组 => useCallback只会在首次渲染时返回缓存函数,后续使用该缓存函数
  • 依赖性不写 => 每次组件重新渲染都会创建新的新缓存函数
  • 添加指定依懒性 => 首次渲染时获取缓存函数,后续是指定依赖性数据发生变化时创建新缓存函数,

验证函数是否重新创建

import { useState, useCallback } from 'react';
const set = new Set();

function App() {
  const [num,setNum] = useState(1);

  const handleNum  = useCallback(()=>{
    setNum( num + 1)
  },[num])
  
  // 验证函数是否缓存
  set.add(handleNum)
  console.log(set);
 
  return (
    <div>
      <h2>首页: {num}--{data} </h2>
      <button onClick={handleNum}>点击添加</button>
    </div>
  );
}
export default App;

注意: 缓存之后,缓存函数的内容(数据)都是一个静止状态, 除非你使用根据依懒性进行变化

import React,{ useState, useCallback } from 'react';
function App() {
  const [num,setNum] = useState(1);
  const [data,setData] = useState("数据");
  
  // 防止组件重新渲染,导致方法被重新创建,
  const handleNum  = useCallback(()=>{
    // 函数被缓存导致数据num一致为1
    console.log(123);
    setNum(num + 1)
  },[]) 
  // 正确写法, 根据数据更新
  const handleNum  = useCallback(()=>{
    setNum( num + 1)
  },[num]) 
  
  // 点击hanleNum 的时候 该函数不会重新创建
  const handleData  = useCallback(()=>{
    setData( data + 1)
  },[data])
  
  return (
    <div>
      <h2>首页: {num}--{data} </h2>
      <button onClick={handleData}>handleData点击添加</button>
    </div>
  );
}

useMemo

用于缓存计算结果，避免不必要的重复计算，只在依赖项发生变化时重新计算

  1. const cachedValue = useMemo(calculateValue, dependencies)
  2. useMemo 会执行回调函数并且将执行结果缓存返回。
  3. useMemo 更适合经过函数计算得到一个确定的值,后续不需更改,
    • 根据父组件传递进行计算获取数据,
    • 自身组件根据获取的数据进行计算
  4. 依懒项
    • 添加空数组 useMemo只在首次渲染时执行一次
    • 添加特定依赖项在首次渲染时执行，在依赖项发生变化时重新执行
    • 不书写useMemo每次执行
function App() {
  const [num,setNum] = useState(0);
  let qwe = 10;
  // 函数重新渲染,会重新函数执行浪费性能
  const result = () => {  
     console.log(1123132); 
     let qqq = null;
     for (let index = 0; index < qwe; index++) {
       qqq += index
     }
    return qqq
  }
  const handleNum  =()=>{
    setNum(num + 1)
  }
  // 根据依懒项 更新数据
   const result = useMemo(() => {  
     console.log(1123132); 
     let qqq = null;
     for (let index = 0; index < qwe; index++) {
       qqq += index
     }
    return qqq
  },[])
 
  return (
    <div>
      <h2>首页: {num}- {result()}</h2>
      <button onClick={handleNum}>点击添加</button>
    </div>
  );
}

useReducer

reducer 是在 React 中用于管理状态的一种模式，通常与 useReducer Hook 结合使用(小型仓库),

状态管理中包含着 => 事件处理数据, 处理数据逻辑, 多个数据集中 将它们进行统一管理

(useReducer的目的是将对应数据的处理逻辑集中,同时进行抽离便于管理数据)

  1. useReducer(reducer, initialArg[,init])
    • reducer：用于更新 state 的纯函数
    • initialArg：用于初始化 state 的任意值
    • 可选参数 init：用于计算初始值的函数
  2. 返回一个由两个值组成的数组 [state,dispatch]
    • 当前的 state
    • dispatch函数用于操作reducer函数中数据处理
      • dispatch 函数允许你更新 state 并触发组件的重新渲染
  3. reducer纯函数
    • 初始化不会执行reducer函数
    • 接收当前状态（state）和一个动作（action）作为参数,并返回一个新的状态,它负责根据动作的类型来更新状态。在 React 中，常常使用 Switch 语句来处理不同的动作类型
// 设置一个对象数据,  单个或者多个
const initialState = { count: 0, age: 123 };

// 设置通信函数    
function reducer(prevState, {type,value}) {
  let newState = {...prevState}
  // 返回一个state新对象,不能返回传入的直接对象 
  switch (type) {
    case "add":
       newState.count++
       return newState
      break;
    case "jian":
         newState.count--
         return newState
        break;
    case "xiu":
         newState.count = value
         return newState
        break;
    default:
      return newState
  }
}

function App() {
  const [state, dispatch] = useReducer(reducer, initialState);
  return (
    <div className="box">
      <h2>首页: {state.count}</h2>
		<button onClick={()=>{dispatch({type: 'add'})}}>点击添加</button>
      	<button onClick={()=>{dispatch({type: 'jian'})}}>减</button>
    </div>
  );
}

自定义hook

hooks函数规范是一个以use开头函数,hooks方法 只能在函数组件顶层调用

  1. 逻辑复用： 自定义 Hook 允许您将通用的逻辑封装到单个函数中，并在多个组件中重复使用。这样可以避免代码重复，提高代码的复用性和可维护性。
  2. 代码组织： 自定义 Hook 可以将相关的逻辑组织到一个地方，使代码更加清晰和可读。通过将逻辑抽象到自定义 Hook 中，可以使组件更专注于 UI 的渲染和交互，而将与逻辑相关的代码单独管理。
  3. 提供清晰的接口： 自定义 Hook 可以定义清晰的接口，使其他开发人员更容易理解和使用。通过命名约定和文档化，可以使自定义 Hook 的用途和用法更加明确。
  4. 无状态逻辑： 自定义 Hook 是纯函数，没有组件实例的概念，因此可以更容易进行测试。可以使用单元测试工具对自定义 Hook 进行测试，以确保其行为符合预期。
  5. 访问 React 的核心功能： 自定义 Hook 可以访问 React 的核心功能，如状态管理（useState）、副作用操作（useEffect）、上下文（useContext）等。这使得自定义 Hook 具有更大的灵活性和功能。
import { useState, useEffect } from 'react';
function useCounter(initialCount, step) {
  const [count, setCount] = useState(initialCount);
  const increment = () => {  setCount(count + step); };
  const decrement = () => {  setCount(count - step);  };
  return [count, increment, decrement];
}
// -------------------------------------
import useCounter from './useCounter';
function CounterComponent() {
  const [count, increment, decrement] = useCounter(0, 1);
  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={increment}>Increment</button>
      <button onClick={decrement}>Decrement</button>
    </div>
  );
}

useTransition

在不阻塞 UI 的情况下更新数据API, 实现UI界面平滑展示

// 在组件调用 useTransition，将某些状态更新标记为 transition(过渡)
const [isPending,startTransition] = useTransition();
  • isPending 否存在待处理的过渡, false不存在过渡, true过渡中
    • 表示过渡是否仍在进行中
  • startTransition 使用此方法将状态更新标记为非阻塞的 transition
    • 在不阻塞 UI 的情况下更新 state 数据
    • startTransition(()=>{}) 传递给 startTransition 的函数必须是同步的。
      • React 会立即不带参数地调用此回调函数，并将在作用域调用期间将所有同步安排的状态更新标记为 transition, 它们将是非阻塞的
      • 回调函数中通过调用一个或多个set 函数更新状态的函数。
      • transition 更新不能用于控制文本输入
import React, { useState, useTransition } from "react";
function Child() { return (  <div> <p>路遥</p> <p>哈哈哈哈</p> </div>  )}

function Base() {
  return (
    <ul>
      {Array(100000)
        .fill("a")
        .map((item, index) => {
          return <li key={index}>{item}</li>;
        })}
    </ul>
  );
}
function App() {
  const [tab, setTab] = useState('about');
  const [isPending, startTransition] = useTransition();
  function selectTab(nextTab) {
    startTransition(() => {
      // 将tab标记为过渡属性: 渲染时不影响UI界面
      setTab(nextTab);
    });
    // setTab(nextTab);
  }

  return (
    <div>
      { 
      	// 先点击效果在点击测试 在通过设置是否设置过渡效果可看出不同
      }
      <button onClick={() => selectTab('contact')}>测试</button>
      <button onClick={() => selectTab('contact1')}>效果</button>
      <hr />
      {tab == "about" ?    <p>123123123</p> : null   }
      {tab == "contact" ?    <Child /> : null   }
      {tab == "contact1" ? <Base /> : null }
    </div>
  );
}

useDeferredValue

延迟更新 UI 的某些部分,提高页面展示流畅度

// 调用 useDeferredValue 来获取该值的延迟版本   value是需要进行延迟的UI部分
const deferredValue = useDeferredValue(value)
  • value：你想延迟的值，可以是任何类型 => 最好传递原始值（如字符串和数字）
    • 在组件的初始渲染期间，返回的延迟值将与你提供的值相同。
    • 但是在组件更新时，React 将会先尝试使用旧值进行重新渲染（因此它将返回旧值），然后再在后台使用新值进行另一个重新渲染（这时它将返回更新后的值）
import React, { useState,useDeferredValue } from "react";

function App() {
  const [text, setText] = useState("");
  const deferredText = useDeferredValue(text);
  return (
    <>
      {
          // 快速输入的过程中 没有延迟更新会很卡
      }
      <input value={text} onChange={(e) => setText(e.target.value)} />
      <ul className="items">
        {Array(10000)
          .fill("a")
          .map((item, index) => {
            return <li key={index}>{deferredText}</li>;
            // 会有很明显的卡顿 
            return <li key={index}>{text}</li>;
          })}
      </ul>
    </>
  );
}

类组件生命周期(不讲)

  • 数据初始
    • constructor(){} state数据初始化
    • static getDerivedStateFromProps(props, state)
      • 会在调用 render 方法之前调用，并且在初始挂载及后续更新时都会被调用,它应返回一个对象来更新 state，如果返回 null 则不做任何操作,
    • render(){} 处理虚拟DOM数据
    • componentDidMount(){} 组件挂载完成之后执行
  • 更新
    • static getDerivedStateFromProps(state,props){}
      • 是否修改 state 数据，如果返回 null 则不处理任何内容
    • shouldComponentUpdate(nextProps, nextState){}
      • 返回值默认为 true; 是否允许页面重新渲染
    • render 渲染数据
    • getSnapshotBeforeUpdate(prevProps, prevState) 获取更新之前的值
    • componentDidUpdate(){} 更新页面完成之后触发
  • 卸载
    • componentWillUnmount()
    • 会在组件卸载及销毁之前直接调用。
    • 在此方法中执行必要的清理操作，例如，清除 timer，取消网络请求或清除在componentDidMount() 中创建的订阅等。
