非阻塞+IO多路复用

1.非阻塞

socket都是会阻塞的(等待状态)

在等待客户端连接以及接收客户端的消息的时候会发生阻塞状态(accept,recv)

设置非阻塞:socket对象.setblocking(False)

服务端代码

import socket
import time
import pprint

# 创建socket对象
server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
# 绑定ip以及端口号
server.bind(("127.0.0.1", 9999))
# 设置监听模式
server.listen(5)
# 设置非阻塞状态的套接字
server.setblocking(False)
# 保存客户端的连接对象的列表
conn_list = []
# 要断开的连接的客户端列表
del_list = []

while True:
    # 捕获阻塞异常
    try:
        # 等待客户端连接
        conn, addr = server.accept()
        conn_list.append(conn)
    except BlockingIOError:
        # 发生BlockingIOError的异常,从列表中获取已经保存的客户端,进行消息接收
        # 从列表中获取连接成功的客户端的对象
        for conn in conn_list:
            try:
                # 接收客户端的消息
                data = conn.recv(1024)
                if not data:
                    conn.close()
                    del_list.append(conn)
                    continue
                print(f"客户端{conn}点菜:{data.decode('utf-8')}")
                conn.send(f"成功收到客户端点的菜{data.decode('utf-8')}.马上准备".encode('utf-8'))
            except BlockingIOError:
                # 如果当下客户端没有消息,就循环其他的客户端对象
                continue
            except ConnectionResetError:
                # 客户端强制断开 意外断开
                conn.close()
                del_list.append(conn)

        for conn in del_list:
            print("断开的conn", conn)
            conn_list.remove(conn)
        del_list.clear()  # 清空del_list
# 关闭套接字对象
server.close()

客户端代码

import socket

client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

client.connect(("127.0.0.1", 9999))

while True:
    msg = input("点菜:")
    if not msg:
        print("请重新点菜")
        continue
    if msg.upper() == "Q":
        print("不点了")
        break
    client.send(msg.encode("utf-8"))

    data = client.recv(1024)
    print(f"服务端的回应:{data.decode('utf-8')}")

client.close() # 四次挥手 相当于发了空数据

2.IO多路复用

IO多路复用是用于提升效率，单个进程可以同时监听多个网络连接IO（一个服务端可以连接多个客户端）

IO多路复用 ，通过一种机制， 可以监视多个文件描述符 ， 一旦描述符就绪(读 ， 写)，能够通知程序进行相应的读写操作

r , w , x = select.select(rlist , wlist , xlist)

rlist -- wait until ready for reading
wlist -- wait until ready for writing
xlist -- wait for an "exceptional condition"
# rlist wlist xlist均是列表
# rlist等待读描述
# wlist等待写描述
# xlist异常
# wlist xlist目前不需要使用到，直接传入一个空列表即可

服务端server

import time

import select
import socket

# 创建socket对象
server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
# ip可复用
server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
# 绑定ip以及端口号
server.bind(("127.0.0.1", 9999))
# 设置监听模式
server.listen(5)
# 设置非阻塞状态的套接字
server.setblocking(False)
# 保存socket对象(第一个数据服务端自己的socket对象,第一个客户端的连接对象conn)
inputs = [server] # [server,conn1,conn2,conn3]

while True:
    r, w, x = select.select(inputs, [], [],0.5) # 监测客户端发来的请求
    print(r)
    # r = [server]
    # r = [conn1]
    # r = [server,conn1]
    # r = [conn1,conn2]
    for read in r:
        # server :有客户端来连接
        if read == server:
            conn, addr = server.accept() # 接收新连接
            print(f"与{addr}连接成功")
            inputs.append(conn)
        else: # conn : 连接成功的客户端中有客户发来消息了
            # 接收消息
            data = read.recv(1024)
            if data:
                print(data.decode('utf-8'))
                read.send(data)
            else:
                print("关闭连接")
                read.close()
                inputs.remove(read)
