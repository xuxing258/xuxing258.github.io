@login_required
def backend(request):
    # 获取到当前用户,通过用户获取到个人站点
    user_obj = models.UserInfo.objects.filter(username=request.user.username, is_delete=False).first()
    blog = user_obj.blog
    # 获取当前用户的所有文章
    article_queryset = models.Article.objects.filter(blog=blog, is_delete=False)
    # 获取当前用户的所有分类
    category_queryset = models.Category.objects.filter(blog=blog, is_delete=False)
    # 获取当前用户的所有标签
    tag_queryset = models.Tag.objects.filter(blog=blog, is_delete=False)
    # 获取当前用户的所有评论
    comment_queryset = models.Comment.objects.filter(user=user_obj)
    return render(request, 'backend/backend.html', locals())

from bs4 import BeautifulSoup

@login_required
def add_article(request):
    user_obj = models.UserInfo.objects.filter(username=request.user.username).first()
    blog = user_obj.blog
    # blog = models.Blog.objects.filter(userinfo__username=request.user.username)

    # 获取当前登录用户的所有分类
    category_queryset = models.Category.objects.filter(blog=blog, is_delete=False)
    # 获取当前登录用户的所有标签
    tag_queryset = models.Tag.objects.filter(blog=blog, is_delete=False)

    if request.method == 'POST':
        title = request.POST.get('title')
        content = request.POST.get('content')
        category_id = request.POST.get('category_id')
        tags = request.POST.getlist('tags')
        # 利用模块处理获取到的content
        soup = BeautifulSoup(content, 'html.parser')
        # 获取简介:从文章内容中切取150字符
        desc = soup.text[0: 150]

        # 删除内容中的script标签
        script_tags = soup.find_all('script')
        for i in script_tags:
            i.decompose()

        # 操作数据库保存文章
        article_obj = models.Article.objects.create(title=title,
                                                    desc=desc,
                                                    content=str(soup),
                                                    category_id=category_id,
                                                    blog=blog)
        # 操作文章和标签的对应关系
        # *tags:将tags列表的每个值变成位置参数传给函数
        article_obj.tags.add(*tags)
        return redirect('/backend/')  # 也可以重定向到首页
    return render(request, 'backend/add_article.html', locals())
