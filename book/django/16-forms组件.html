from django.shortcuts import render, HttpResponse
from app01.myforms import regform

def index(request):
    # 1.产生空的对象
    form_obj = regform.RegForm()
    # 3.校验数据
    if request.method == 'POST':
        # 数据校验时的对象名要和第一个空的对象名一致
        # 用户输入过的内容会被value属性接收不论get请求还是post请求都应该要展示一样的input框
        form_obj = regform.RegForm(request.POST)
        # 4.判断数据是否符合校验规则
        if form_obj.is_valid():  # 只有当数据全部符合校验规则时返回True
            return HttpResponse('注册成功')
        # 5.数据不合法,返回前端页面展示提示信息
        
    # 2.通过locals()将当前函数内部的名字传递给前端页面
        
    return render(request, 'index.html', locals())

添加更多校验信息

  • 钩子函数
# 钩子函数:在特定的节点自动触发完成响应
# 类似于forms组件中的第二道校验规则(通过了默认的校验规则时自动触发),能够让我们自定义更多校验规则

# 在forms组件中有两类钩子:
# 1.局部钩子:当你在给单个字段增加校验规则的时候可以使用
    # 局部钩子:判断用户名是否存在
    # 局部钩子函数名:clean_字段名
    def clean_username(self):
        # 1.钩到要校验的数据
        # 但凡来到钩子函数,说明通过了第一道校验规则,cleaned_data将会保存所有通过校验的键值对
        username = self.cleaned_data.get('username')
        
        # 2.检验数据,不符合校验规则的添加提示信息
        is_exists = models.User.objects.filter(username=username).exists()
        if is_exists:  # 用户存在
            self.add_error('username', '用户已存在')
            # add_error(字段, 提示信息):给指定字段添加提示信息
            
        # 3.将钩出来的数据返回
        return username

# 2.全局钩子:当前你需要给多个字段添加校验规则时使用
	# 全局钩子:校验两次密码是否一致
    def clean(self):
        # 1.勾出需要校验的数据
        password = self.cleaned_data.get('password')
        re_password = self.cleaned_data.get('re_password')
        
        # 2.校验数据是否合法, 不符合添加提示信息
        if password != re_password:  # 不相等
            self.add_error('re_password', '两次密码不一致')
            
        # 3.将勾出来的数据进行返回
        # 勾出来的数据太多的情况下,可以直接返回整个字典
        return self.cleaned_data

from django import forms 
from django.core.validators import RegexValidator, EmailValidator

class RegForm(forms.Form):
    ....
	# validators:添加django的校验规则
    # RegexValidator(正则表达式, 不通过校验时的提示信息)
    phone = forms.CharField(label='手机号', max_length=11, min_length=11,
                            error_messages={
                                'max_length': '手机号最大11位',
                                'min_length': '手机号最小11位',
                                'required': '手机号不能为空'
                            },
                            validators=[
                                RegexValidator('\d+', '请输入数字'),
                                RegexValidator('1[3-9]\d{9}', '请输入正确格式的手机号')
                            ])
                            # 1[3-9]\d{9}
                            # 第一位取1
                            # [3-9]:表示从3-9这些数字中取其中的一个
                            # \d:取任意的数字,同[0-9]的效果一样
                            # {9}: 取前面的字符九次
                        
    # EmailValidator('提示信息'):用来校验是否符合邮箱格式                   
	email2 = forms.CharField(max_length=32, label='邮箱',
                             validators=[
                                 EmailValidator('邮箱格式不正确'),
                                 RegexValidator('\w+@\w+\.\w+', message='邮箱格式不正确')
                             ])

其他参数补充

label:指定字段名
error_messages:自定义报错信息(只能给字段的属性自定义报错)
required:控制字段是否必填(默认值为True)
initial:默认值
