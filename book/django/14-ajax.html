def author_add(request):
    if request.method == 'POST':
        name = request.POST.get('name')
        age = request.POST.get('age')
        sex = request.POST.get('sex')
        email = request.POST.get('email')
        addr = request.POST.get('addr')

        # 先产生作者详情对象,再传递给作者进行添加
        author_detail_obj = models.AuthorDetail.objects.create(email=email,
                                                               addr=addr)
        models.Author.objects.create(name=name,
                                     age=age,
                                     sex=sex,
                                     author_detail=author_detail_obj)

        return redirect('author_list')
    return render(request, 'author_add.html')

def author_edit(request, edit_id):
    edit_obj = models.Author.objects.filter(id=edit_id).first()
    if request.method == 'POST':
        name = request.POST.get('name')
        age = request.POST.get('age')
        sex = request.POST.get('sex')
        email = request.POST.get('email')
        addr = request.POST.get('addr')
        models.Author.objects.filter(id=edit_id).update(name=name, age=age, sex=sex)
        models.AuthorDetail.objects.filter(author__id=edit_id).update(email=email, addr=addr)
        return redirect('author_list')
    return render(request, 'author_edit.html', locals())

choices参数

针对某个可以列举完全可能性字段,可以利用choices参数进行保存

# 在models.py的模型类中通过元祖列举所有可能性
# 然后再通过choices参数指定这个元祖
    gender_choices = (
        (1, '男'),
        (2, '女'),
        (3, '保密')
    )
    gender = models.IntegerField(choices=gender_choices, null=True)

    score_choices = (
        (1, '优秀'),
        (2, '良好'),
        (3, '及格'),
        (4, '不及格')
    )
    score = models.IntegerField(choices=score_choices, null=True)
    
    
# 1.gender字段存的数字不在元祖的列举的范围中是可以保存的
# 2.获取整型对应的值:get_字段名_display()
# 如果整型在元祖的范围内,可以获取到对应的值
# 不在范围内,获取的还是数字
res = models.User.objects.filter(id=1).first()
print(res.gender)  # 1
print(res.get_gender_display())  # 男
print(res.score)  # 1
print(res.get_score_display())  # 优秀

ajax

submit:提交post请求,但是会刷新整个页面
    
ajax:异步提交,局部刷新
    能够动态获取数据之后跟后端确认并实时展示到前端,整个过程不刷新页面
    
# 注意:在form表单中的button默认类型也是submit,所在在form表单中写按钮时,需要手动指定类型
<form action="" method="post">
	...
	<button type="button">点我</button> 
</form>
  • 朝后端提交请求的方式
1.浏览器地址栏输入url回车--->get请求
2.a标签的href属性--->get请求
3.form表单--->get请求/post请求
4.ajax--->get请求/post请求

ajax基本格式

 $.ajax({
            // 1.朝哪个地址发送请求
            url: '', // 空的:朝当前所在地址提交;路由:朝指定路由提交 
            // 2.请求方式
            type: 'post', // 默认为get请求
            // 3.发送的数据:以字典的合适发送
            data: {
                ...
            },
            // 4.请求发送后端执行成功之后, 自动调用执行函数
            // args:后端执行成功之后的返回值会被args接收
            success: function (args) {
                ...
}

def index(request):
    if request.method == 'POST':
        i1 = request.POST.get('i1')
        i2 = request.POST.get('i2')
        i3 = int(i1) + int(i2)
        return HttpResponse(i3)
    return render(request, 'index.html', locals())

<script>
    // 通过点击button朝后端提交ajax请求
    // click(function(){}):点击事件
    $('#btn').click(function () {
        // 发送ajax请求
        $.ajax({
            // 1.朝哪个地址发送请求
            url: '', // 空的:朝当前所在地址提交;路由:朝指定路由提交 
            // 2.请求方式
            type: 'post', // 默认为get请求
            // 3.发送的数据
            data: {
                // val():括号内不传参表示获取input框内文本
                'i1': $('#i1').val(),
                'i2': $('#i2').val()
            },
            // 4.请求成功之后调用的函数
            // args:后端执行成功之后的返回值会被args接收
            success: function (args) {
                // val():括号内传参表示给input框传文本
                $('#i3').val(args)
            }
        })
    })
</script>
