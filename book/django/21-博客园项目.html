# 完整实现
<script>
    // change():改变事件,随着input框内容发生变化执行
    // 必须等待file框选择完文件之后再改变img标签的src属性
    $('#myfile').change(function () {
        // 1.生成一个文件阅读器对象
        let fileReader = new FileReader();
        // 2.获取文件框的文件对象
        let file = $('#myfile')[0].files[0];
        // 3.利用文件阅读器读取文件对象
        fileReader.readAsDataURL(file) 
        // 读取需要时间,并不会等待读取完毕就会立即执行后面修改的操作
        // 等待文件阅读器读取完毕在做修改的操作
        fileReader.onload = function () {
            // 4.修改img标签的src属性
            $('#myimg').attr('src', fileReader.result)
        }
    })
</script>
  • 并发
计算机同时有多个进行进程在同时执行,但其实并不是真正的在同时运行
而是操作系统快速轮换调用不同的进程,使得看起来像是在同时运行
(两个进程:
 第一个进行遇到了time.sleep(1),获取网络访问,需要等待
 那么就会调用第二个进程执行,等第二个进程执行完毕或者同样的遇到了等待
 就会切换回第一个进程执行
)

        $.each([1, 2, 3, 4, 5], function(index, obj){
            console.log(index, obj)
        })

        0 1
        1 2
        2 3
        3 4
        4 5

def register(request):
    # 1.生成空对象
    form_obj = RegForm()

    # 3.post请求接收前端传递过来的数据
    if request.method == 'POST':
        # 返回字典
        back_dict = {'code': 0, 'msg': ''}
        form_obj = RegForm(request.POST)
        if form_obj.is_valid():  # 所有字段都通过校验才会返回True
            # 通过校验的数据
            clean_data = form_obj.cleaned_data  # username, password, confirm_password, email, phone
            # 删除confirm_password键值对
            clean_data.pop('confirm_password')
            # 获取用户上传的头像对象
            avatar = request.FILES.get('avatar')
            if avatar:  # 如果获取到了保存到clean_data字典中, 如果没有,使用数据库的默认值
                clean_data['avatar'] = avatar
            # 操作数据库保存数据
            # **字典:将字典变成关键字参数传给函数
            models.UserInfo.objects.create_user(**clean_data)
            # 给返回的字典添加url键
            back_dict['url'] = '/login/'

        else:  # 数据不符合校验规则
            back_dict['code'] = 2000
            back_dict['msg'] = form_obj.errors

        # 4.返回字典给前端页面进行判断
        return JsonResponse(back_dict)

    # 2.传给html页面渲染html标签
    return render(request, 'register.html', locals())

$.ajax({
    ...
    success: function (args) {
    // args:正确的情况下:{code: 0, msg: '', url: '/login/'}
    // 错误的情况下:{code: 2000, msg: {报错信息}}
    if(args.code===0){
        // 跳转到url这个键指定的地址
        window.location.href = args.url
    }else{
        // 展示报错信息到指定的span标签内
        // msg:{'报错的字段': ['报错的信息'], ....}
        // 循环遍历args.msg获取的索引是字典的键, 值是字典的值
        $.each(args.msg, function (index, obj) {
            // index: 报错的字段--->username
            // obj:[对应的报错信息]--->['用户名已存在']
            // id_index--->id_username
            let target_id = '#id_' + index  // 获取到报错的input框的id
            // parent():获取到当前操作的标签的父标签
            // addClass(类):给当前操作的标签加上一个类
            $(target_id).next().text(obj[0]).parent().addClass('has-error')
        })
    }
}
