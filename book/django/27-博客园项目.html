注册功能(补充)

之前写的时候没有给用户添加对应的blog,是去到后台管理页面进行手动添加.

后续相关的添加文章,分类和标签都是依赖于blog表的对象.如果用户没有绑定blog,那么将无法添加对应的信息.所以在写注册功能时,同时添加对应的blog

# views.py
# 在注册功能的视图函数中添加
def register(request):
    # 1.生成空对象
    form_obj = RegForm()

    # 3.post请求接收前端传递过来的数据
    if request.method == 'POST':
        # 返回字典
        back_dict = {'code': 0, 'msg': ''}
        form_obj = RegForm(request.POST)
        if form_obj.is_valid():  # 所有字段都通过校验才会返回True
            # 通过校验的数据
            clean_data = form_obj.cleaned_data  # username, password, confirm_password, email, phone
            # 删除confirm_password键值对
            clean_data.pop('confirm_password')
            # 获取用户上传的头像对象
            avatar = request.FILES.get('avatar')
            if avatar:  # 如果获取到了保存到clean_data字典中, 如果没有,使用数据库的默认值
                clean_data['avatar'] = avatar
            # 操作数据库保存数据
            # **字典:将字典变成关键字参数传给函数
            # 添加个人站点
            username = clean_data.get('username')
            blog_obj = models.Blog.objects.create(site_name=username)

            models.UserInfo.objects.create_user(blog=blog_obj, **clean_data)
            # 给返回的字典添加url键
            back_dict['url'] = '/login/'

        else:  # 数据不符合校验规则
            back_dict['code'] = 2000
            back_dict['msg'] = form_obj.errors

        # 4.返回字典给前端页面进行判断
        return JsonResponse(back_dict)

    # 2.传给html页面渲染html标签
    return render(request, 'register.html', locals())

@login_required
def article_edit(request, article_id):
    user_obj = models.UserInfo.objects.filter(username=request.user.username).first()
    blog = user_obj.blog
    # 通过接收到的文章id获取到文章对象
    article_obj = models.Article.objects.filter(id=article_id,
                                                blog=blog,
                                                is_delete=False).first()

    # 获取当前登录用户的所有分类
    category_queryset = models.Category.objects.filter(blog=blog, is_delete=False)
    # 获取当前登录用户的所有标签
    tag_queryset = models.Tag.objects.filter(blog=blog, is_delete=False)
    if request.method == 'POST':
        title = request.POST.get('title')
        content = request.POST.get('content')
        category_id = request.POST.get('category_id')
        tags = request.POST.getlist('tags')
        # 利用模块处理获取到的content
        soup = BeautifulSoup(content, 'html.parser')
        # 获取简介:从文章内容中切取150字符
        desc = soup.text[0: 150]

        # 删除内容中的script标签
        script_tags = soup.find_all('script')
        for i in script_tags:
            i.decompose()

        # 操作数据库保存文章
        models.Article.objects.filter(id=article_id).update(title=title,
                                                            desc=desc,
                                                            content=str(soup),
                                                            category_id=category_id)
        # 操作文章和标签的对应关系
        article_obj.tags.set(tags)
        return redirect('/backend/')  # 也可以重定向到首页
    return render(request, 'backend/article_edit.html', locals())

@login_required
def user_edit(request):
    back_dict = {'code': 0, 'email': '', 'phone': ''}
    if request.method == 'POST':
        username = request.POST.get('username')
        email = request.POST.get('email')
        phone = request.POST.get('phone')
        site_name = request.POST.get('site_name')
        site_title = request.POST.get('site_title')
        site_theme = request.FILES.get('site_theme')
        # 修改userinfo表数据
        models.UserInfo.objects.filter(username=request.user.username).update(username=username,
                                                                              email=email,
                                                                              phone=phone)
        # 修改blog表数据
        blog_obj = models.Blog.objects.filter(userinfo__username=request.user.username).first()
        blog_obj.site_name = site_name
        blog_obj.site_title = site_title
        blog_obj.site_theme = site_theme
        blog_obj.save()
        return JsonResponse(back_dict)
