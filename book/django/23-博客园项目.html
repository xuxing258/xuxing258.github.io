from django.contrib.auth.decorators import login_required

@login_required
def set_password(request):
    if request.method == 'POST':
        # 针对ajax请求,都返回一个字典信息
        back_dict = {'code': 0, 'msg': ''}
        old_password = request.POST.get('old_password')
        new_password = request.POST.get('new_password')
        re_password = request.POST.get('re_password')
        # 1.判断旧密码是否正确
        is_right = request.user.check_password(old_password)
        if is_right:
            # 2.判断两次密码是否一致
            if new_password == re_password:
                # 3.修改密码
                request.user.set_password(new_password)
                request.user.save()
                back_dict['msg'] = '修改成功'
            else:
                back_dict['code'] = 1001
                back_dict['msg'] = '两次密码不一致'
        else:
            back_dict['code'] = 1002
            back_dict['msg'] = '旧密码错误'
        return JsonResponse(back_dict)

media文件夹配置
	该配置可以让用户上传的所有文件都固定放在某一个指定的文件夹下
 
1.在根目录下创建media文件夹,将之前创建的avatar文件夹和site_theme文件夹直接复制到media文件夹中
2.去settings.py中配置media文件夹的路径
    # media文件夹配置
    MEDIA_ROOT = BASE_DIR / 'media'
    
    # 或者用os模块拼接路径
    import os
	MEDIA_ROOT = os.path.join(BASE_DIR, 'media')

3.将media的路径暴露给外部获取,去urls.py中配置
from django.urls import path, re_path
from django.views.static import serve
from bbs import settings

urlpatterns = [
    ...
    # media路由
    # .*:接收零个或任意多个的任意字符
    re_path('media/(?P<path>.*)', serve, {'document_root': settings.MEDIA_ROOT})
]

from django.db.models import Count
from django.db.models.functions import TruncMonth

# 个人站点
def site(request, username):
    # 先校验用户名所对应的个人站点是否存在
    user_obj = models.UserInfo.objects.filter(username=username).first()

    if not user_obj:
        return render(request, 'error.html')

    blog = user_obj.blog

    # 查询当前个人站点下的所有文章
    article_list = models.Article.objects.filter(is_delete=False, blog=blog)

    # 1.获取当前个人站点的分类以及分类下的文章数
    category_list = (models.Category.objects.filter(blog=blog, is_delete=False).
                     annotate(count_num=Count('article__id')).
                     values_list('name', 'count_num'))

    # 2.获取当前个人站点的标签以及标签下的文章数
    tag_list = (models.Tag.objects.filter(blog=blog, is_delete=False).
                annotate(count_num=Count('article__id')).
                values_list('name', 'count_num'))

    # 3.获取当前个人站点的分布月份,以及不同月份下的文章数
    # 对所以的文章进行分组,然后通过TruncMonth('create_time'),得到一个新的字段month获取到年和月
    # 再通过values('month')获取到刚刚切取出来的字段,对这个字段进行分组,
    # annotate(count_num=Count('id'))统计出这个分组下的文章的数量,用count_num新字段接收
    # values_list('month', 'count_num')):获取到month字段以及count_num字段的值返回值为元祖
    date_list = (article_list.annotate(month=TruncMonth('create_time')).
                 values('month').annotate(count_num=Count('id')).
                 values_list('month', 'count_num'))

    return render(request, 'site.html', locals())
