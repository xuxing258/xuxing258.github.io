drf框架

视图

  • APIView:在django的View的基础上增加了免除csrf,认证,权限,限流功能
  • GenericAPIView:继承APIView,增加了一些功能方法,将数据库查询,序列化类提取到类变量中
  • ViewSetMixin:将post/get/put/patch/delete方法映射到了list/create/retrieve/update/partial_update/destroy
# urls.py
path('users/', views.UserView.as_view({'get': 'list', 'post': 'create'}))
path('users/<int:pk>', views.UserView.as_view({'get': 'retrieve', 'put': 'update', 'patch': 'partial_update', 'delete': 'destroy'}))

  • GenericViewSet:继承GenericAPIView和ViewSetMixin

  • ListModelMixin, CreateModelMixin, RetrieveModelMixin, DestroyModelMixn, UpdateModelMixin:

    在五个类中帮我们写好了list, create, retrieve, upadte, partial_update, destroy方法.继承五大类和GenericViewSet之后,只需要在类中定义类变量queryset, serializer_class即可

条件搜索

如果某个接口需要传递一些条件进行搜索,就是在url后面通过问号传参即可

自定义Filter

from rest_framework import serializers
from app01 import models
from rest_framework.viewsets import ModelViewSet
from rest_framework.filters import BaseFilterBackend


class UserModelSerializer(serializers.ModelSerializer):
    level_text = serializers.CharField(source='get_level_display', read_only=True)

    class Meta:
        model = models.UserInfo
        fields = ['username', 'age', 'email', 'level_text']


# 自定义过滤器
class MyFilter(BaseFilterBackend):
    def filter_queryset(self, request, queryset, view):
        # 查询年龄=多少的数据
        age = request.query_params.get('age')
        if not age:
            return queryset
        return queryset.filter(age=age)

# 自定义过滤器
class MyFilter2(BaseFilterBackend):
    def filter_queryset(self, request, queryset, view):
        # 查询id小于多少的数据
        user_id = request.query_params.get('id')
        if not user_id:
            return queryset
        return queryset.filter(id__lt=user_id)


class UserView(ModelViewSet):
    queryset = models.UserInfo.objects.all()
    serializer_class = UserModelSerializer
	# 在类中通过类变量引用定义好的过滤器
    # 视图内部在获取列表,获取单条数据,更新,删除都会执行
    filter_backends = [MyFilter, MyFilter2]

    def perform_create(self, serializer):
        serializer.save(password='123456')
        
        
# 访问:
get请求:http://localhost:8000/users/?id=3 --->查询id小于3的数据
       http://localhost:8000/users/?age=20 ---?查询age=20的数据

  • lookup_expr
'exact': 等于
'iexact': 等于(忽略大小写)
'contains': 包含
'icontains': 包含(忽略大小写)
'startswith': 以某个字符开头
'endswith': 以某个字符结尾
'gt': 大于
'gte': 大于等于
'le':小于
'lte': 小于等于
'in':判断在多个值中的某一个
'range': 获取某个范围的数据
'isnull': 获取为空的数据
    
    
class MyFilterSet(FilterSet):
    # ?depart=部门名字--->查询部门名字为某个的用户
    depart = filters.CharFilter(field_name='depart__name', lookup_expr='exact')
    # ?min_id=数字 --->查询id大于等于某个数字的用户
    min_id = filters.NumberFilter(field_name='id', lookup_expr='gte')
    # name=某个字符 --->查看username中是否包含某个字符的用户
    name = filters.CharFilter(field_name='username', lookup_expr='contains')
    # ?level=1&level=2 --->查看level等于1和level等于2的数据(多个条件同时满足)
    level = filters.MultipleChoiceFilter(field_name='level', lookup_expr='exact', choices=models.UserInfo.level_choices)
    # age=19,20 ---> age in [19, 20]
    age = filters.BaseInFilter(field_name='age', lookup_expr='in')
    # range_id_max=10&range_id_min=1 ---> id between 1 and 10
    range_id = filters.NumericRangeFilter(field_name='id', lookup_expr='range')
    # ?ordering=id--->根据id升序进行排序
    # ?ordering=-id--->根据id降序进行排序
    # ?ordering=age--->根据age升序进行排序
    # ?ordering=-age--->根据age降序进行排序
    ordering = filters.OrderingFilter(fields=['id', 'age'])
    # 自定义函数进行搜索:?size=1
    size = filters.CharFilter(method='filter_size')
    
    class Meta:
        model = models.UserInfo
        fields = ['depart', 'min_id', 'name', 'level', 'age', 'range_id', 'ordering']
        
    def filter_size(self, queryset, name, value):
        int_value = int(value)
        return queryset[0: int_value]

分页

如果数据量比较大,不能把所以数据都展示给用户看,需要通过分页进行展示

# BasePagination:分页类父类
# PageNumberPagination(BasePagination):?page=2&page_size=30
# LimitOffsetPagination(BasePagination):?offset=10&limit=10
# CursorPagination(BasePagination):支持上一页和下一页的格式进行分页

APIView

如果视图类继承APIView,那么在使用分页时,需要手动实例化和调用相关方法

  • PageNumberPagination
# settings.py
REST_FRAMEWORK = {
    'PAGE_SIZE': 2  # 默认每页显示2条数据
}


# urls.py
path('users/', views.UserView.as_view()),


# views.py
from rest_framework.views import APIView
from rest_framework.pagination import PageNumberPagination


class UserModelSerializer(serializers.ModelSerializer):
    level_text = serializers.CharField(source='get_level_display', read_only=True)

    class Meta:
        model = models.UserInfo
        fields = ['username', 'age', 'email', 'level_text']


class UserView(APIView):
    def get(self, request):
        queryset = models.UserInfo.objects.all()

        # 进行分页
        page = PageNumberPagination()
        pagination_queryset = page.paginate_queryset(queryset, request, self)
        ser = UserModelSerializer(instance=pagination_queryset, many=True)
        return Response({'code': 0, 'data': ser.data})
    
    
# 访问
get请求: http://localhost:8000/users/?page=2

from rest_framework.pagination import CursorPagination
from rest_framework.views import APIView
from rest_framework import serializers
from rest_framework.response import Response
from app01 import models

class UserModelSerializer(serializers.ModelSerializer):
    level_text = serializers.CharField(source='get_level_display', read_only=True)

    class Meta:
        model = models.UserInfo
        fields = ['username', 'age', 'email', 'level_text']
        
        
class MyCursor(CursorPagination):
    ordering = 'id'
    page_size_query_param = 'size'
    page_size = 2  # 不传size默认显示2条, 做多100条
    max_page_size = 100
    
    
class UserView(APIView):
    def get(self, request):
        queryset = models.UserInfo.objects.all().order_by('id')

        # 进行分页
        page = MyCursor()
        pagination_queryset = page.paginate_queryset(queryset, request, self)
        ser = UserModelSerializer(instance=pagination_queryset, many=True)
        return page.get_paginated_response({'code': 0, 'data': ser.data})
