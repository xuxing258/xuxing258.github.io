拖拽

(抓个bo)draggable属性允许元素进行拖拽,并分为两个部分

  • 源对象：指的是我们鼠标点击的一个事物，这里可以是一张图片，一个DIV，一段文本等等。

  • 目标对象：指的是我们拖动源对象后移动到一块区域，源对象可以进入这个区域，可以在这个区域上方悬停(未松手)，可以释松手释放将源对象放置此处(已松手)，也可以悬停后离开该区域

    • 元素默认是不可以拖拽的,图片默认可以拖拽;

    draggable属性

<div id="box" draggable=true> </div>

// true  规定元素课拖动
// false 规定元素不可拖动
// auto  使用浏览器默认行为

拖拽源事件

  • ondragstart 用户开始拖动元素时触发事件

  • ondrag 元素正在拖动时触发事件

  • ondragend 用户完成元素拖动后触发事件

拖拽目标

  • ondragenter - 当源对象进入目标对象容器范围内时触发此事件
  • ondragover - 源对象在目标容器范围内拖动时触发此事件 持续触发
  • ondragleave - 当源对象离开目标对象容器范围内时触发此事件
  • ondrop - 当鼠标在目标区域抬起时触发 默认情况下,浏览器会阻止这个事件的触发
    • 需要在dragover事件中阻止默认行为
    • 在目标容器的 dragover 事件处理器中，调用 event.preventDefault()，以使得该元素能够接收 drop 事件
let element = null;
box.ondragstart=function(event){
  // 保存节点
  element = event.target;
}

box1.ondragover=function(event){
  // 阻止浏览器不允许触发 drop事件
  event.preventDefault()
}

box1.ondrop=function(){
  this.appendChild(element)
}

H5 提供了存储了拖拽时初始数据 ,只能存储字符串并且只会在ondrop抬起的时候能取值到

存在event.dataTransfer中

event.dataTransfer.setData("存储名称","存储的值")  // 存值
event.dataTransfer.getData("存入名称")  // 取值

// h5传入的参数 只会在ondrop抬起的时候能取值到

box.ondragstart=function(event){
	event.dataTransfer.setData("key",event.target.className)
}

box1.ondragover=function(event){
  // 阻止浏览器不允许触发 drop事件
  event.preventDefault()
}

box1.ondrop=function(event){
  let dom = document.getElementsByClassName(event.dataTransfer.getData("key"))[0]
  this.appendChild(dom)
}

注意: 源对象也可以是目标对象

拖拽上传图片

<!DOCTYPE html>
<html>
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>moban</title>
  <style>
    body,h1,h2,h3,h4,h5,h6,ul,ol,p{margin:0;font-weight: normal}
    ul,ol{padding:0; list-style:none;}
    a{text-decoration: none;}
    #box{
      width: 1000px;
      height: 600px;
      background: #ccc;
      margin: 50px auto 0;
    }
    .wrap{
      width: 1000px;
      margin: 0px auto;
    }
  </style>
</head>
<body>
 
  <div id="box">请把图片拖拽至改区域</div>
  <div class="wrap">
    <p>当前图片<span class="item">0</span>张,多少<span class="kb">0</span>M</p>
    <input type="file">
  </div>

  <script>
    let box = document.getElementById("box")
    let item = document.getElementsByClassName("item")[0]
    let kb = document.getElementsByClassName("kb")[0]
    let arr = []; // 保存上传图片的每一项
    box.addEventListener("dragover",(ev)=>{
      // 去除浏览器的默认事件=> 不会触发drop事件 
      ev.preventDefault()
      if(box.innerText == "请把图片拖拽至改区域"){
        box.innerText = ""
      }
    })
    box.addEventListener("drop",(ev)=>{
      // 去除浏览器的默认事件 => 默认打开图片
      ev.preventDefault()
      // 拖拽会将默认数据放在 event.dataTransfer.files
      let files = ev.dataTransfer.files
      // 循环每一项
      appendFile(files)
      // 计算函数
      computed()
    })

    function appendFile(files){
      for(let i=0;i<files.length;i++){
        // 生成blod   参数需要是数组格式的
        console.log(files[i]);
        let blob = new Blob([files[i]]);
        // 生成blod 链接
        arr.push(files[i])
        let url = URL.createObjectURL(blob);
        console.log(url);
        // 创建图片
        let oImg = new Image()
        console.log(url);
        oImg.src = url; // 异步代码
        oImg.width=200;
        oImg.height = 200;
        oImg.onload=function(){
          box.appendChild(oImg)
        }
      }
    }

    function computed(){
      item.innerText = arr.length
      let result  = arr.reduce((pecu,next)=>pecu + next.size,0)/1024/1024
      // 数值.toFixed(参数)  保留几位数
      kb.innerText = result.toFixed(2);
    }
  </script>
</body>
</html>
